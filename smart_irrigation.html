The Smart Irrigation System automates the watering of plants by monitoring soil moisture levels using a soil moisture sensor. When the soil becomes dry, the system turns on a water pump, and when the soil is moist enough, it automatically turns off the pump. This helps conserve water and ensures optimal irrigation for crops.

üîß Components

  • Arduino UNO
  • Soil Moisture Sensor
  • Relay Module
  • Water Pump
  • Jumper Wires
  • Power Supply

üíª Code

const int sensor_pin = A0;
const int relay_pin = 7;
int moisture_value = 0;

void setup() {
  Serial.begin(9600);
  pinMode(relay_pin, OUTPUT);
}

void loop() {
  moisture_value = analogRead(sensor_pin);
  Serial.print("Soil Moisture Value: ");
  Serial.println(moisture_value);

  if (moisture_value < 400) {
    digitalWrite(relay_pin, HIGH); // Turn ON pump
    Serial.println("Soil is Dry - Pump ON");
  } else {
    digitalWrite(relay_pin, LOW);  // Turn OFF pump
    Serial.println("Soil is Wet - Pump OFF");
  }
  delay(2000);
}
